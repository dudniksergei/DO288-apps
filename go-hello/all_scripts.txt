#
# S2I assemble script for the 's2i-do288-go' image.

#
# For more information refer to the documentation:

#

if [[ "$1" == "-h" ]]; then
  # If the 's2i-do288-go' assemble script is executed with '-h' flag,

  exec /usr/libexec/s2i/usage
